class Solution {
public:
    vector<int> getRow(int rowIndex) {
        vector<vector<int>> ans;
        for (int i = 0; i <= rowIndex; i++) {
            vector<int> current;
            for (int j = 0; j <= i; j++) {
                if (j == 0 || j == i)
                    current.push_back(1);
                else
                    current.push_back(ans[i-1][j-1]+ans[i-1][j]);
            }
            ans.push_back(current);
        }
        return ans[rowIndex];
    }
};
